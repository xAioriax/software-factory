Tatetí

Juego clásico de tatetí en un tablero 3x3 para dos jugadores locales.

Turno de X
0 Partidas ganadas por X
0 Partidas perdidas por X

Instrucciones: toca una casilla vacía para colocar tu marca. Gana quien complete tres iguales en línea horizontal, vertical o diagonal. Si se llena el tablero sin ganador, la partida termina empatada. Los contadores de ganadas y perdidas para X se guardan en este navegador.

const STORAGE_KEY = 'tateti-contadores-x'; const tableroEl = document.getElementById('tablero'); const estadoEl = document.getElementById('estado'); const ganadasEl = document.getElementById('ganadas'); const perdidasEl = document.getElementById('perdidas'); const reiniciarBtn = document.getElementById('reiniciar'); const borrarBtn = document.getElementById('borrar'); let casillas = Array(9).fill(''); let turno = 'X'; let partidaTerminada = false; let contadores = cargarContadores(); function cargarContadores() { try { const guardado = localStorage.getItem(STORAGE_KEY); if (!guardado) return { ganadas: 0, perdidas: 0 }; const datos = JSON.parse(guardado); return { ganadas: Number.isFinite(datos.ganadas) ? datos.ganadas : 0, perdidas: Number.isFinite(datos.perdidas) ? datos.perdidas : 0 }; } catch (error) { return { ganadas: 0, perdidas: 0 }; } } function guardarContadores() { localStorage.setItem(STORAGE_KEY, JSON.stringify(contadores)); } function actualizarContadores() { ganadasEl.textContent = contadores.ganadas; perdidasEl.textContent = contadores.perdidas; } function renderizarTablero() { tableroEl.innerHTML = ''; casillas.forEach((valor, indice) => { const boton = document.createElement('button'); boton.className = 'celda' + (valor ? ' ' + valor.toLowerCase() : ''); boton.type = 'button'; boton.textContent = valor; boton.setAttribute('aria-label', valor ? `Casilla ${indice + 1}, ocupada por ${valor}` : `Casilla ${indice + 1}, vacía`); boton.disabled = partidaTerminada || valor !== ''; boton.addEventListener('click', () => jugar(indice)); tableroEl.appendChild(boton); }); } function checkWin(jugador) { const combinaciones = [ [0, 1, 2], [3, 4, 5], [6, 7, 8], [0, 3, 6], [1, 4, 7], [2, 5, 8], [0, 4, 8], [2, 4, 6] ]; return combinaciones.some(([a, b, c]) => casillas[a] === jugador && casillas[b] === jugador && casillas[c] === jugador); } function hayEmpate() { return casillas.every(casilla => casilla !== ''); } function terminarPartida(mensaje, clase) { partidaTerminada = true; estadoEl.className = 'estado ' + (clase || ''); estadoEl.textContent = mensaje; renderizarTablero(); } function jugar(indice) { if (partidaTerminada || casillas[indice] !== '') return; casillas[indice] = turno; renderizarTablero(); if (checkWin(turno)) { if (turno === 'X') { contadores.ganadas += 1; guardarContadores(); actualizarContadores(); terminarPartida('¡Ganó X! Nueva partida ganada para X.', 'ganador'); } else { contadores.perdidas += 1; guardarContadores(); actualizarContadores(); terminarPartida('¡Ganó O! Partida perdida para X.', 'perdedor'); } return; } if (hayEmpate()) { terminarPartida('Empate: nadie ganó esta partida.', ''); return; } turno = turno === 'X' ? 'O' : 'X'; estadoEl.className = 'estado'; estadoEl.textContent = 'Turno de ' + turno; } function reiniciarPartida() { casillas = Array(9).fill(''); turno = 'X'; partidaTerminada = false; estadoEl.className = 'estado'; estadoEl.textContent = 'Turno de X'; renderizarTablero(); } function borrarContadores() { contadores = { ganadas: 0, perdidas: 0 }; guardarContadores(); actualizarContadores(); localStorage.removeItem(STORAGE_KEY); reiniciarPartida(); } reiniciarBtn.addEventListener('click', reiniciarPartida); borrarBtn.addEventListener('click', borrarContadores); actualizarContadores(); renderizarTablero();
